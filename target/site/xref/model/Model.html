195 	public Permeability checkBump(int x, int y) {
196 
197 		Permeability permBump = null;
198 		for (IMobile obj2 : Armobile) {
199 
200 			if ((obj2.getX() == x) && (obj2.getY() == y)) {
201 				if (obj2.getPerm() == Permeability.MONSTER) {
202 					this.permBump = Permeability.MONSTER;
203 					return this.permBump;
204 				} else if (obj2.getPerm() == Permeability.CHARACTER) {
205 					this.permBump = Permeability.CHARACTER;
206 
207 				} else if (obj2.getPerm() == Permeability.SPELL) {
208 					this.permBump = Permeability.SPELL;
209 				}
210 			}
211 		}
212 		for (IElements obj : Arimages) {
213 
214 			if ((obj.getX() == x) && (obj.getY() == y)) {
215 				if (obj.getPerm() == Permeability.COLLECTABLE) {
216 					this.permBump = Permeability.COLLECTABLE;
217 					System.out.println("Score +100");
218 				} else if (obj.getPerm() == Permeability.PENETRABLE) {
219 					this.permBump = Permeability.PENETRABLE;
220 				} else if (obj.getPerm() == Permeability.BLOCKING) {
221 					this.permBump = Permeability.BLOCKING;
222 				}
223 			}
224 		}
225 		return this.permBump;
226 
227 	}
